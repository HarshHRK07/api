HRK's AI API Docs

GET /openai/gpt

Python Integration:


import requests

url = 'https://green-devil.tech/openai/gpt'
params = {'prompt': 'Hello'}

response = requests.get(url, params=params)
print(response.json())

cURL Integration:


curl https://green-devil.tech/openai/gpt?prompt=Hello

Next.js Integration:


import axios from 'axios';

const handleSubmit = async (prompt) => {
  const response = await axios.get('https://green-devil.tech/openai/gpt', {
    params: {
      prompt
    }
  });

  console.log(response.data);
}

Node.js Integration:


const https = require('https');

const options = {
  hostname: 'green-devil.tech',
  path: '/openai/gpt?prompt=Hello',
  method: 'GET'
};

const req = https.request(options, res => {
  console.log(`statusCode: ${res.statusCode}`);

  res.on('data', d => {
    process.stdout.write(d);
  });
});

req.on('error', error => {
  console.error(error);
});

req.end();

Java Integration:


import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;

public class OpenAIGPTRequest {
    public static void main(String[] args) throws Exception {
        String urlString = "https://green-devil.tech/openai/gpt?prompt=Hello";
        URL url = new URL(urlString);
        HttpURLConnection con = (HttpURLConnection) url.openConnection();
        con.setRequestMethod("GET");

        int responseCode = con.getResponseCode();
        if (responseCode == HttpURLConnection.HTTP_OK) {
            BufferedReader in = new BufferedReader(new InputStreamReader(con.getInputStream()));
            String inputLine;
            StringBuilder response = new StringBuilder();
            while ((inputLine = in.readLine()) != null) {
                response.append(inputLine);
            }
            in.close();
            System.out.println(response.toString());
        } else {
            System.out.println("GET request failed: " + responseCode);
        }
    }
}

GET /ytdl/mp3

Python Integration:


import requests

url = 'https://green-devil.tech/ytdl/mp3'
params = {'url': 'https://www.youtube.com/watch?v=dQw4w9WgXcQ'}

response = requests.get(url, params=params)
with open('video.mp3', 'wb') as f:
    f.write(response.content)
print('Download complete!')

cURL Integration:


curl https://green-devil.tech/ytdl/mp3?url=https://www.youtube.com/watch?v=dQw4w9WgXcQ -o video.mp3

Next.js Integration:


import axios from 'axios';

const handleDownload = async (videoUrl) => {
  const response = await axios.get('https://green-devil.tech/ytdl/mp3', {
    params: {
      url: videoUrl
    },
    responseType: 'arraybuffer'
  });

  const blob = new Blob([response.data], { type: 'audio/mpeg' });
  const url = window.URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'video.mp3';
  a.click();
}
